//NUMBER GUESSING GAME
//GENERATE RANDOM NUMBERS BETWEEN 1 AND 10 ASK USER TO GUESS TILL THEY GET IT RIGHT

START
    SET RANDOM_NUMBER TO RANDINT(1,10)
    SET RUNNING TO TRUE
    WHILE RUNNING LOOP
    DISPLAY "ENTER A NUMBER TO GUESS"
    INPUT NUMBER
        IF NUMBER = RANDOM_NUMBER THEN
            DISPLAY "HOORAY YOU WIN"
            DISPLAY "GAME END"
            SET RUNNING TO FALSE
        END IF
        ELSE
            DISPLAY "WRONG TRY AGAIN"
        END IF
    END LOOP
END
